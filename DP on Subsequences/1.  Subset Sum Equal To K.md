#### 1.Recursive Approach 
```
def subsetSumToKH(n,k,arr,dp):
    if n <= 0:
        # If k = 0, we reached target sum
        if k == 0:
            return True
        else:
            return False

    if arr[n-1] <= k:
        include = subsetSumToKH(n-1, k-arr[n-1], arr,dp)
        exclude = subsetSumToKH(n-1, k, arr,dp)
        if include or exclude:
            return True
        else:
            return False
    else:
        return subsetSumToKH(n-1, k, arr,dp)
    
def subsetSumToK(n, k, arr):
    dp = [[False for _ in range(k+1) for _ in range(n+1)]]
    return subsetSumToKH(n,k,arr,dp)
```

#### 2.Memoization

```
def subsetSumToK(n, k, arr):
    dp = [[-1 for _ in range(k+1)] for _ in range(n+1)]
    return subsetSumToKH(n,k,arr,dp)

def subsetSumToKH(n,k,arr,dp):
    if n <= 0:
        # If k = 0, we reached target sum
        if k == 0:
            return True
        else:
            return False

    if dp[n][k] != -1:
        return dp[n][k]


    if arr[n-1] <= k:
        dp[n][k] = subsetSumToKH(n-1, k-arr[n-1], arr,dp) or subsetSumToKH(n-1, k, arr,dp)
        return dp[n][k]

    else:
        dp[n][k] = subsetSumToKH(n-1, k, arr,dp)
        return dp[n][k]
```

#### 3.Iterative
```
def subsetSumToK(N, s, arr):
    dp = [[False for _ in range(s+1)] for _ in range(N+1)]
    for i in range(N+1):
        for j in range(s+1):
            if i == 0:
                dp[i][j] = False
            if j == 0:
                dp[i][j] = True   
            else:    
                if arr[i-1] <= j:
                    dp[i][j] = dp[i-1][j-arr[i-1]] or dp[i-1][j]
                else:
                    dp[i][j] = dp[i-1][j]
    return dp[N][s]
```
