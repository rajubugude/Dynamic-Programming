def frogJump(n, heights, i): #recursion
    if i == n-1:
        return 0

    one = abs(heights[i + 1] - heights[i]) + frogJump(n, heights, i + 1)
    two = abs(heights[i + 2] - heights[i]) + frogJump(n, heights, i + 2)
    ans = min(one,two)
    return ans
